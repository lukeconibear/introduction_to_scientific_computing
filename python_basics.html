Functions¶

def say_hello(*args): # arguments
    for arg in args:
        print(f'Hello {arg}!')

say_hello('Gary')

Hello Gary!

say_hello('Gary', "Gary's Mother")

Hello Gary!
Hello Gary's Mother!

def do_maths(x=2, y=3, z=4): # keyword arguments
    print(f'x = {x}')
    print(f'y = {y}')
    print(f'z = {z}')
    result = (x + y) * z
    print(f'result = {result}')

do_maths()

x = 2
y = 3
z = 4
result = 20

do_maths(1, 2, 3)

x = 1
y = 2
z = 3
result = 9

do_maths(3, 2, 1)

x = 3
y = 2
z = 1
result = 5

do_maths(z=3, y=2, x=1)

x = 1
y = 2
z = 3
result = 9

def do_maths(**kwargs): # keyword arguments
    print(kwargs)
    return sum(kwargs.values())

result = do_maths(x=2, y=3, z=4)

{'x': 2, 'y': 3, 'z': 4}

result
